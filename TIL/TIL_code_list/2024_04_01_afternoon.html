다양한 방법으로 DOM에 접근해보기

<!DOCTYPE html> <html lang="en"> <head> <style> </style> </head> <body> <div class="animal"> <h1>개</h1> <h1>고양이</h1> <h1>말</h1> </div> <h1 id="lion">사자</h1> <h1>쥐</h1> <h1 class="sheep">양</h1> <script> //querySelector는 클래스의 이름을 사용한 여러 요소 중에서 첫 번째 요소만 반환한다. let animal = document.querySelector(".animal h1"); animal.style.color = "green"; //배열접근 불가능 //n[1].style.color = "red"; //querySelectorAll은 클래스의 이름을 사용한 여러 요소들을 배열로 받아온다. let animals = document.querySelectorAll(".animal h1"); //배열접근 가능 animals[1].style.color = "red"; animals[2].style.color = "blue"; //id를 통해서 DOM요소에 접근 let lion = document.getElementById("lion"); lion.style.color = "green"; //태그 이름을 통해서 DOM요소에 접근 animals = document.getElementsByTagName("h1"); //DOM 요소를 이용해서 style 적용(배열 사용) animals[4].style.color = "orange"; //클래스명을 통해서 DOM요소에 접근 sheep = document.getElementsByClassName("sheep"); //DOM 요소를 이용해서 style 적용(배열 사용) sheep[0].style.color = "pink"; </script> </body> </html>

innerText, innerHTML 프로퍼티를 활용해서 현재 시간 표시하기

<!DOCTYPE html> <html lang="en"> <head> <style> </style> </head> <body> <button onclick="inntext()">innerText로 표시하기</button> <button onclick="innhtml()">innerHTML로 표시하기</button> <h1>현재 시각:</h1> <div id="current"></div> <script> let now; //text 내용만 표시한다 function inntext(){ now = new Date(); document.getElementById('current').innerText = now; } //text 내용과 html 태그까지 적용해서 표시한다 function innhtml(){ now = new Date(); let clockbutton = document.getElementById('current'); clockbutton.innerHTML = "<em>" + now + "</em>"; //콘솔창에 html 태그까지 표시됨 console.log(clockbutton.innerHTML); //콘솔창에 thml 태그는 표시되지 않음 console.log(clockbutton.innerText); } </script> </body> </html>
